/*----------------------------------------------------------------------*\
|The initial function used to determine the command type and then sends  |
|it to a function that will handle it.                                   |
\*----------------------------------------------------------------------*/
 
void getCommand(char *argv[]){
    struct PROCESS Process;
    char command[256];
    int Quit = FALSE;
 
    while(Quit == FALSE){
    fgets(command,sizeof(command),stdin);
        if (command[0] == 'q'){
            printf("Program terminated\n");
            Quit = TRUE;
        }else if (command[0] == '+'){
            if (checkFormat(argv, command) == TRUE){
                addProcess(command, argv);
            }
        }else if (command[0] == '*'){
            if (checkFormat(argv, command) == TRUE){
                updatePriority(command, argv);
            }
        }else if (command[0] == '?'){
            if (checkFormat(argv, command) == TRUE){
                getProcess(command, argv);
            }
        }else if (command[0] == 'p'){
            if (checkFormat(argv, command) == TRUE){
                printCsv(argv, command);
            }
        }else if (command[0] == 'r'){
            if (checkFormat(argv, command) == TRUE){
                printRecordIndex(argv);
            }
        }else if (command[0] == 'b'){
            if (checkFormat(argv, command) == TRUE){    
                printBucketIndex(argv);
            }
        }else if (command[0] == 'h'){
            if (checkFormat(argv, command) == TRUE){
                help();
            }
        }else{
            printf( "Command not recognized. Please try again.\n");
        }
    }
}
 
/*---------------------------------------------------------------------------*\
|The AddProcess() function is called by the getCommand() function. It is used |
|used to add a new process to the process.bin  file. It makes use of a        |
|tokenizer to determine the entered values that followed the '+' symbol.      |
|This function makes use of the getBucketLocation() function along with       |
|appendToProcess() and existInBucket(). *side note* does not currently handle |
|pId values less than 0 and greater than 999 *side note end*                  |
\*---------------------------------------------------------------------------*/
 
void addProcess(char command[], char *argv[]){
    struct PROCESS Process;
    struct RECORD Record;
    struct TEMP Temp;
 
    char *sp = " \t";
    char *token;
    char *ptr = command;
    int bucketfd = 0, recordfd = 0, processesfd = 0, num = 0, loopCount = 0;
    int doneSearch = FALSE;
    int bucketLocation;
 
    token = strtok(ptr, sp);
    Process.processId = atoi(strtok(NULL, sp));
    strcpy(Process.owner, strtok(NULL, sp));
    strcpy(Process.programName, strtok(NULL, sp));
    Process.totalTime = atoi(strtok(NULL, sp));
    Process.timeLeft = atoi(strtok(NULL, sp));
    Process.priority = atoi(strtok(NULL, sp));
   
    bucketLocation = getBucketLocation(argv, Process.processId);
    if (bucketLocation == EMPTY){
        appendToProcess(argv, Process);
        printf("ADD: %d,%s,%s,%d,%d,%d\n",
                Process.processId, Process.owner, Process.programName,
                Process.totalTime, Process.timeLeft,Process.priority);
    }else{
        if( existInBucket(argv, Process.processId, bucketLocation) == TRUE){
            printf("ERROR -- Already Exists\n");
        }else{
            appendToProcess(argv, Process);
            printf("ADD: %d,%s,%s,%d,%d,%d\n",
                    Process.processId, Process.owner, Process.programName,
                    Process.totalTime, Process.timeLeft,Process.priority);
        }
    }
}
 
/*--------------------------------------------------------------------------*\
|The updatePriority() function is called by the the getCommand() function. It|
|is used to update the priority of a process stored in the process.bin file. |
|This function makes use of a temp struct in order to store Process values   |
|temporarily so that they may be used in the future.                         |
\*--------------------------------------------------------------------------*/
 
void updatePriority(char command[],char *argv[]){
    struct PROCESS Process;
    struct RECORD Record;
    struct TEMP Temp;
    int sequenceNum;
    int processesfd = 0, num = 0;
   
    char *sp = " \t";
    char *token;
    char *ptr = command;
   
    token = strtok(ptr, sp);
    Process.processId = atoi(strtok(NULL, sp));
    Temp.newPriority = atoi(strtok(NULL, sp));
   
    sequenceNum = searchForProcess(argv, Process.processId);
    if (sequenceNum == EMPTY){
        printf("Error -- Process does not exist\n");
    }else{
        processesfd = open(argv[3], O_RDWR|O_CREAT, S_IRWXU);
        lseek(processesfd, sequenceNum * sizeof(Process), 0);
        num = read (processesfd, &Process, sizeof(Process));
        lseek(processesfd, sequenceNum * sizeof(Process), 0);    
       
        Temp.oldPriority = Process.priority;
        Process.priority = Temp.newPriority;
       
        num = write(processesfd, &Process, sizeof(Process));
        printf("UPDATE: %d, old: %d, new:%d\n",
                Process.processId, Temp.oldPriority, Process.priority);
    }
    close(processesfd);
}

/*------------------------------------------------------------------------*\
|PrintCsv() is a function that writes all inputted processes to a csv file |
|with a name of the users choice.                                          |
\*------------------------------------------------------------------------*/
 
void printCsv(char *argv[], char command[]){
    struct PROCESS Process;
    struct stat statBuffer;
    int csvfd = 0, processesfd = 0, fileLeft = 0, num = 0;
    char fname[128];
    char empty[] = "File empty";
    int BUFFER_SIZE = 1024;
    char *sp = " \t\n";
    char *token;
    char *ptr = command;
 
    token = strtok(ptr, sp);
    strcpy(fname, strtok(NULL, sp));
    strcat(fname,".csv");
    processesfd = open(argv[3], O_RDONLY);
    fileLeft = fstat(processesfd, &statBuffer);
    close(processesfd);
    fileLeft = statBuffer.st_size;
 
    if (fileEmpty(argv, 3) == TRUE){
        char *buffer = (char*) malloc(BUFFER_SIZE);
        csvfd = open(fname, O_TRUNC|O_RDWR|O_CREAT, S_IRWXU);
        num = snprintf(buffer, BUFFER_SIZE, "%s\n", empty);
        num = write(csvfd, buffer, num);
        close(csvfd);
    }else{
        csvfd = open(fname, O_TRUNC|O_RDWR|O_CREAT, S_IRWXU);
        processesfd = open(argv[3], O_RDONLY);
       
        char *buffer = (char*) malloc(BUFFER_SIZE);
        num = read(processesfd, &Process, sizeof(Process));
        while(fileLeft != 0){
            num = snprintf(buffer, BUFFER_SIZE, "%d, %s, %s, %d, %d, %d\n",
                    Process.processId, Process.owner, Process.programName,
                    Process.totalTime, Process.timeLeft,Process.priority);
            num = write(csvfd, buffer, num);
            fileLeft = fileLeft - sizeof(Process);
            num = read(processesfd, &Process, sizeof(Process));
        }
        close(processesfd);
        close(csvfd);
    }
    printf("file written\n");
}
 
/*-------------------------------------------------------------------*\
|This function grabs the bucket location from the "bucket_index.bin"  |

|these functions performs a similar task, however, they are used in   |
|in different situations. This function is to be used when checking   |

int getBucketLocation(char *argv[], int pId){
    struct RECORD Record;
    int bucketNum;

    int bucketfd = 0, num = 0;

    bucketfd = open(argv[1], O_RDONLY);

/*--------------------------------------------------------------------------*\
|The searchForProcess() function is used to find a sequence number           |
|that can then be used to read or edit a given process. This function stream |
|lines the search by using the index's rather than scanning each indevidual  |
|process.                                                                    |
\*--------------------------------------------------------------------------*/
 
int searchForProcess(char *argv[], int pId){
    struct PROCESS Process;
    int processesfd = 0, num = 0, loopCount = 0;
    int result;
    int bucketLocation;
    int sequenceNum;
   
    bucketLocation = getBucketLocation(argv, pId);
   
    if (bucketLocation == EMPTY){
        result = EMPTY;
    } else {
        sequenceNum = getSequenceNum(argv, pId, bucketLocation);
        if (sequenceNum == EMPTY){
            result = EMPTY;
        }else{
            result = sequenceNum;
        }
    }
    return result;
}
 
/*--------------------------------------------------------------------------*\

void appendToProcess(char *argv[], struct PROCESS Process){
    struct RECORD Record;
    int processesfd = 0, num = 0;
   
    processesfd = open(argv[3], O_RDWR|O_APPEND|O_CREAT, S_IRWXU);
    num = write(processesfd, &Process, sizeof(Process));
    close(processesfd);
 
    addToBucket(argv,Process.processId);
}
 
/*---------------------------------------------------------------------------*\
|ExistInBucket() is a function that uses the buckets location and searches the|
|record_index.bin file to see if the given process Id matches with any in the |
|bucket. If it does exist, the function will return true, if not, it will     |
|return false.                                                                |
\*---------------------------------------------------------------------------*/
 
int existInBucket(char *argv[], int pId,  int bucketLocation){
    struct RECORD Record;
    int doneSearch = FALSE;
    int result;
    int recordfd = 0, num = 0, loopCount = 0;
 
    recordfd = open(argv[2], O_RDONLY);
    lseek(recordfd, bucketLocation * sizeof(Record), 0);
    num = read(recordfd, &Record, sizeof(Record));
    close(recordfd);
   
    while( doneSearch == FALSE && loopCount < BUCKET_MAX){
        if( Record.processId[loopCount] == pId){
            result = TRUE;
            doneSearch = TRUE;
        }else if (Record.processId[loopCount] == EMPTY){
            result = FALSE;
            doneSearch = TRUE;
        }else{
            loopCount++;
        }
    }
    return result;
}
 
/*-------------------------------------------------------------------*\
|This function does the same thing as getProcessLocation, however,    |
|this function is to be used when the user is confident that there is |
|a path to the process that uses the indexes. The other function      |
|takes more of a brute force approach while this one uses the index's.|
\*-------------------------------------------------------------------*/
 
int getSequenceNum(char *argv[], int pId, int bucketLocation){
    struct RECORD Record;
    int doneSearch = FALSE;
    int sequenceNum;
 
    int recordfd = 0, num = 0, loopCount = 0;
   
    recordfd = open(argv[2], O_RDONLY);
    lseek(recordfd, bucketLocation * sizeof(Record), 0);
    num = read(recordfd, &Record, sizeof(Record));
    close(recordfd);
   
    while( doneSearch == FALSE && loopCount < BUCKET_MAX){
        if( Record.processId[loopCount] == pId){
            sequenceNum = Record.sequenceNum[loopCount];
            doneSearch = TRUE;
        }else if (Record.processId[loopCount] == EMPTY){
            sequenceNum = EMPTY;
            doneSearch = TRUE;
        }else{
        loopCount++;
        }
    }
    return sequenceNum;
}
 
/*-----------------------------------------------------------------------*\
|AddToBucket() is a function that either adds a new process to a record   |
| bucket or creates a new bucket if none is currently present. To create  |
|the new bucket it makes call to createBucket().                          |
\*-----------------------------------------------------------------------*/
 
void addToBucket(char *argv[], int pId){
    struct PROCESS Process;
    struct RECORD Record;
    int processLocation;
    int doneSearch = FALSE;
    int bucketLocation;
    int recordfd = 0, num = 0, loopCount = 0;
 
    bucketLocation = getBucketLocation(argv, pId);
    processLocation = getProcessLocation(argv, pId);
 
    if (bucketLocation == EMPTY){
        createBucket(argv, pId, processLocation);
    }else{
        recordfd = open(argv[2], O_RDWR|O_CREAT, S_IRWXU);
        lseek(recordfd, bucketLocation * sizeof(Record), 0);
        num = read(recordfd, &Record, sizeof(Record));
        while(doneSearch == FALSE && loopCount < BUCKET_MAX){
            if(Record.processId[loopCount] == pId){
                printf("ERROR -- Record already exists");
                doneSearch = TRUE;
            }else if (Record.processId[loopCount] == EMPTY){
                Record.processId[loopCount] = pId;
                Record.sequenceNum[loopCount] = processLocation;
                lseek(recordfd, bucketLocation * sizeof(Record), 0);        
                num = write(recordfd, &Record, sizeof(Record));
                doneSearch = TRUE;
            }else{
                loopCount++;
            }
        }
        close(recordfd);
    }
    addToBuckLocation(argv, pId);
}
 
/*---------------------------------------------------------------------------*\
|AddToBuckLocation() is a simple function that adds the location of a bucket  |
|to the bucket_index.bin file.                                                |
\*---------------------------------------------------------------------------*/
 
void addToBuckLocation (char *argv[], int pId){
    struct RECORD Record;
    int bucketNum;
    int bucketLocation;
    int recordLocation;
    int bucketfd = 0, num = 0;
   
    bucketNum = pId/BUCKET_MAX;
    recordLocation = getRecordLocation(argv, pId);
    bucketfd = open(argv[1], O_RDWR);
    lseek(bucketfd, bucketNum * sizeof(int), 0);
    num = write(bucketfd, &recordLocation, sizeof(int));
    close(bucketfd);
}
 
/*---------------------------------------------------------------------------*\
|The getProcessLocation() function does the same task as the getSequenceNum   |
|function, however, it does so in a brute force approach. It is to be used    |
|when a new process is just entered and a record is being created. In this    | 
|case this function will be used to scan the processes.bin file and find the  |
| location.                                                                   |
\*---------------------------------------------------------------------------*/
 
int getProcessLocation(char *argv[], int pId){
    struct PROCESS Process;
    int processLocation;
    int searchPId;
    int doneSearch = FALSE;
    int processesfd = 0, num = 0, loopCount = 0;
   
    searchPId = pId;
    processesfd = open(argv[3], O_RDONLY);
    while (doneSearch == FALSE){
        num = read(processesfd, &Process, sizeof(Process));
        if(Process.processId == searchPId){
            processLocation = loopCount;
            doneSearch = TRUE;
        }else{
            loopCount++;
        }
    }
    close(processesfd);
    return processLocation;
}
 
/*---------------------------------------------------------------------------*\
|The getRecordLocation() function is the brute force approach to finding      |
|the location of a bucket in the record_index.bin file. This is used when     |
|a bucket is present but the location is not set in the bucket_index.bin      |
|file. In that case, the bucket must be found with a brute force approach     |
|such as this.                                                                |
\*---------------------------------------------------------------------------*/
 
int getRecordLocation(char *argv[], int pId){
    struct RECORD Record;
    int recordLocation;
    int doneSearch = FALSE;
    int bucketNum;
    int recordfd = 0, num = 0, loopCount = 0;
   
    bucketNum = pId/BUCKET_MAX;
    recordfd = open(argv[2], O_RDONLY);
    while (doneSearch == FALSE && loopCount < BUCKET_MAX){
        num = read(recordfd, &Record, sizeof(Record));
        if((Record.processId[0]/BUCKET_MAX) == bucketNum ){
            recordLocation = loopCount;
            doneSearch = TRUE;
        }else{
            loopCount++;
        }    
    }
    close(recordfd);
    return recordLocation;
}
 
/*--------------------------------------------------------------------------*\
|CreateBucket is a simple function that makes a new bucket consisting of     |
|all -1's except for the first tupple whose pId and sequenceNum are set to   |
|the passed values.                                                          |
\*--------------------------------------------------------------------------*/
 
void createBucket(char *argv[], int pId, int pLoc){
    struct RECORD Record;
    int recordfd = 0, num = 0, loopCount = 1;
 
    recordfd = open(argv[2], O_RDWR|O_APPEND|O_CREAT, S_IRWXU);
    Record.processId[0] = pId;
    Record.sequenceNum[0] = pLoc;
   
    while (loopCount <10){
        Record.processId[loopCount] = -1;
        Record.sequenceNum[loopCount] = -1;
        loopCount++;
    }
    num = write(recordfd, &Record, sizeof(Record));
    close(recordfd);
}
